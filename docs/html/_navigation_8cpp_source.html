57 
58 std::vector<double> pidOut;
59 //double newVelocity = 0;
60 double errorValue = 0; // to store current error
61 double propOutput = 0; // to store proportional output
62 double integralVal = 0; // temp variable to calculate integral value
63 double intOutput = 0; // to store integral output
64 double derivativeVal = 0; // temp variable to calculate derivative value
65 double derOutput = 0; // to store derivative output
66 
67 errorValue = setPoint - currentVelocity;
68 propOutput = kp_ * errorValue;
69 
70 integralVal += errorValue * diffTime_;
71 intOutput = ki_ * integralVal;
72 
73 derivativeVal = ((errorValue - previousError_) / diffTime_);
74 derOutput = kd_ * derivativeVal;
75 
76 pidOut.push_back(errorValue);
77 pidOut.push_back(propOutput);
78 pidOut.push_back(intOutput);
79 pidOut.push_back(derOutput);

81 }

128 double Navigation::calculate(double targetHeading,
129  double currentVelocity,
130  double setPoint,
131  int flag) {
132 double newVelocity = 0;
133 double tempVel; // temp variable to store current velocity
134 int velocityConverged = 0;
135 int headingConverged = 0;
136 std::vector<double> pidOutput;
137 
138 double outTime = 0; // temp variable to determine convergence time
139 SteerAlgorithm Ackermann = SteerAlgorithm();
140 
141 double tempHeading = targetHeading; // to store the target for graph
142 
143 // std::vector<std::pair<double, double>> combinedXY(1, std::make_pair(0, 0));
144 std::vector<std::pair<double, double>> points;
145 std::vector<std::pair<double, double>> pointsVelocity;
146 
147 // while loop to run until velocity and heading are converged
148 while(velocityConverged != 1 && headingConverged != 1) {
149  // Run PID control initially
150  pidOutput = calculatePID(setPoint, currentVelocity);
151  newVelocity = currentVelocity +
152  (pidOutput[1] + pidOutput[2] + pidOutput[2]);
153  previousError_ = pidOutput[0];
154  currentVelocity = newVelocity;
155 
156  clock_t toc = clock();
157 
158  // store the current values in a vector for plotting
159  pointsVelocity.push_back(std::make_pair((
160  static_cast<double> (toc)/CLOCKS_PER_SEC),
161  currentVelocity));
162  if (flag == 2) {
163  // show the values on screen
164  std::cout << "Current Velocity: " << currentVelocity
165  << " Setpoint: " << newVelocity << std::endl;
166  }
167  if (setPoint == newVelocity) {
168  velocityConverged = 1;
169  }
170  if ((targetHeading-heading) != 0 && velocityConverged == 1) {
171  if (newVelocity > maxTurnVelocity) {
172  tempVel = maxTurnVelocity;
173  }
174  tempVel = newVelocity;
175  if (targetHeading-heading > 0) {
176  dir = 1;
177  } else {
178  dir = -1;
179  }
180  // change wheel angles to move robot towards target
181  Ackermann.changeWheelAngles(Ackermann.getCorrRadius_(),
182  shaftLength,
183  shaftDistance);
184 
185  double arcLen = Ackermann.arcLength(targetHeading,
186  Ackermann.getCorrRadius_());
187  auto turnTime = Ackermann.turnTime(arcLen, newVelocity);
188  clock_t tic = clock();
189 
190  // loop until heading is converged towards target
191  while ((outTime < turnTime) && (targetHeading != heading)) {
192  pidOutput = calculatePID(setPoint, currentVelocity);
193  newVelocity = currentVelocity +
194  (pidOutput[1] + pidOutput[2] + pidOutput[2]);
195  previousError_ = pidOutput[0];
196  currentVelocity = newVelocity;
197 
198  clock_t toc = clock();
199  outTime = (static_cast<double> (toc - tic)) / CLOCKS_PER_SEC;
200  double newArcLen = tempVel * outTime;
201  heading = (newArcLen * targetHeading / arcLen);
202 
203  // store the current values in a vector for plotting
204  points.push_back(std::make_pair(
205  (static_cast<double> (toc)/CLOCKS_PER_SEC),
206  heading));
207  if (flag == 2) {
208  std::cout << "Current Velocity: " << currentVelocity
209  << " Setpoint: " << newVelocity << std::endl;
210  std::cout << "Current Heading: " << heading
211  << " Target: " << targetHeading << std::endl;
212  }
213  }
214  targetHeading = 0;
215  headingConverged = 1;
216  Ackermann.resetWheel();
217  }
218 }
219 
220 if (flag == 2) {
221  // Plot the graph on gnuplot by calling the below lines
222  gnuSteerAngleGraph(points, tempHeading, true);
223  gnuVelocityGraph(pointsVelocity, newVelocity, true);
224 return 0;
225 } else if (flag == 1) {
226  return newVelocity;
227 } else {
228  return heading;
229 }
230 return newVelocity;
231 }
